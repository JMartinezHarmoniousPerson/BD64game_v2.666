ACTOR 64PortableFirstAidKit : DoomWeapon
{
Scale 0.8
Obituary "How did you die from health. Are you a vampire?"
Tag "First Aid Kit"
Inventory.Icon "WMEDP0"
Weapon.SlotNumber 1
Weapon.SelectionOrder 9999
Weapon.AmmoType1 "64Bandages"
Weapon.AmmoUse1 1
Weapon.AmmoGive1 0
Weapon.AmmoType2 "64AdrenalineStim"
Weapon.AmmoUse2 1
Weapon.AmmoGive2 0
+WEAPON.AMMO_CHECKBOTH
+WEAPON.AMMO_OPTIONAL
+WEAPON.WIMPY_WEAPON
+WEAPON.CHEATNOTWEAPON
+WEAPON.NOALERT
VisibleToPlayerClass "64DoomPlayer"
Inventory.ForbiddenTo "Tactical64DoomPlayer","Tactical64DoomPlayer_p","Modern"
+INVENTORY.RESTRICTABSOLUTELY
+INVENTORY.UNDROPPABLE
	States
	{
	Ready:
		MDKI A 1 A_WeaponReady
		Loop
	Select:
		MDKI A 1 A_Raise
		Wait
	Deselect:
		MDKI A 1 A_Lower
		Wait
	Fire:
		MDKI BCDEFG 2
		TNT1 A 0 A_ReFire
		Goto StopHealAnimation
	Hold:
		TNT1 A 0 A_JumpIfHealthLower(100,"CheckForBandages2")
		MDKI H 1 A_Print("You Are Healthy")
		MDKI H 25
		Goto StopHealAnimation
	CheckForBandages2:
		TNT1 A 0 A_JumpIfInventory("64Bandages",1,"RealHealingAnimation")
		MDKI H 1 A_Print("Not Enough Bandages!")
		MDKI H 25
		Goto StopHealAnimation
	RealHealingAnimation:
		TNT1 A 0 A_PlaySound("firstaidkit/bandaging",1)
		TNT1 A 0 A_Stop
		MDKI H 4 A_TakeInventory("64Bandages",1)
		TNT1 A 0 A_Stop
		MDKI I 4 HealThing(1,100)
		TNT1 A 0 A_Stop
		MDKI J 4
		TNT1 A 0 A_Stop
		TNT1 A 0 A_ReFire
		Goto StopHealAnimation
	StopHealAnimation:
		MDKI GFEDCB 1
		Goto Ready
	AltFire:
		TNT1 A 0 A_JumpIfInventory("64AdrenalineStim",1,"UseBerserkerAdrenaline")
		MDKI A 1 A_Print("You look inside your First Aid Kit for Berserk Syringes and find nothing...")
		MDKI A 25
		Goto Ready
	UseBerserkerAdrenaline:
		MDQ1 ABCDEFGHIJKL 1
		MDQ1 L 2
		MDQ1 M 1 A_TakeInventory("64AdrenalineStim",1)
		TNT1 A 0 A_Stop
		TNT1 A 0 A_PlaySound("firstaidkit/zerk",1)
		TNT1 A 0 HealThing(100,0)
		TNT1 A 0 A_SetBlend("ff0000",0.75,70)
		TNT1 A 0 A_GiveInventory("PowerStrength")
		MDQ1 N 2
		MDQ1 NOPQHGFDCB 1
		MDQ1 A 1 A_FireCustomMissile("EmptySyringeSpawn",-5,0,8,-18)
		Goto Ready
	}
}

ACTOR EmptySyringe
{
Scale 0.5
Height 2
Radius 2
Speed 4
Mass 6
SeeSound "weapons/shell"
-DROPOFF
-NOGRAVITY
+WINDTHRUST
+CLIENTSIDEONLY
+MOVEWITHSECTOR
+MISSILE
+NOBLOCKMAP
+NOTELEPORT
+FORCEXYBILLBOARD
+NOTDMATCH
+GHOST
+ISMONSTER		//INFRARED
	States
	{
	Spawn:
		ZRKS U 0 NoDelay A_Jump(128,2)
		TNT1 A 0 A_SetScale(0-ScaleX,ScaleY)		//FLIP IMAGE HORZ
		ZRKS U 4 A_CheckFloor("Death")
		WAIT
	Death:
		ZRKS W 4
	rest:
		TNT1 A 0 ACS_NamedExecuteAlways("LowBloodCheckerACS")//Check Effects
		ZRKS X 400
		TNT1 A 0 A_JumpIfInventory("LowGraphicsMode",1,"Stoping")
		Loop
	Stoping:
		TNT1 A 1
		Stop
	}
}

	{
	Spawn:
		STIM A 0 NoDelay A_JumpIf(CallACS("CensoredHealing_ACS")==1,"AlphaSpawn")

		Stop
	AlphaSpawn:

		Stop

		Stop
	}
}
