Actor 64Berserk : Berserk Replaces Berserk
{
Scale 0.82
	states
	{
	Spawn:
		PSTR A 0 NoDelay A_JumpIf(CallACS("CensoredHealing_ACS")==1,"CensoredSpawn")	//R31 added option
		TNT1 A 0 A_JumpIf(CallACS("CensoredHealing_ACS")==2,"BFGStyleSpawn")	//R31 added option
		PSTR A 32
		Loop
	CensoredSpawn:
		PSTR B 32
		Goto Spawn
	BFGStyleSpawn:
		PSTR C 32
		Goto Spawn
	Pickup:
		TNT1 A 0 A_SetBlend("ff0000",0.75,70)
		TNT1 A 0 A_GiveInventory("PowerStrength")
		TNT1 A 0 HealThing(100,0)
		TNT1 A 0 A_JumpIf(CallACS("BerserkSwitchFistsACS")==0,"NoSwitch")	//R31 added option
		TNT1 A 0 A_JumpIfInventory("IsTacticalClass",1,"Pickup2")
		TNT1 A 0 A_JumpIfInventory("IsModern",1,"Pickup2")
		TNT1 A 0 A_SelectWeapon("64Fist")
		Stop
	Pickup2:
		TNT1 A 0 A_SelectWeapon("64Fisttactical")
		Stop
	NoSwitch:
		TNT1 A 0
		Stop
	}
}
