Actor 64ChainsawTACTICAL : Brutal64Weapon		//R5 added CHAINSAW FOR TACTICAL CLASS
{
Scale 0.82
//+INVENTORY.ALWAYSPICKUP		//R12.1 removed
Weapon.Kickback 0
Weapon.SelectionOrder 2201		//R21 WAS 1000
Weapon.AmmoUse 0
Weapon.AmmoGive 100
Weapon.AmmoType "SawGas"
Inventory.PickupMessage "$GOTCHAINSAW"
Obituary "$OB_MPCHAINSAW"
Tag "$TAG_CHAINSAW"
+WEAPON.MELEEWEAPON
+WEAPON.AXEBLOOD
+INVENTORY.RESTRICTABSOLUTELY		//R15 added
+NOAUTOFIRE
+NOALERT
	States
	{
	KickingFlash:	//r31 new kick animations
		TNT1 A 0 A_PlaySound("weapons/sawidle",5)
		SAWS RQPON 1
		SAWS D 7
		SAWS NOPQR 1
		Stop
	AirKickingFlash:
		TNT1 A 0 A_PlaySound("weapons/sawidle",5)
		SAWS RQPON 1
		SAWS D 9
		SAWS NOPQR 1
		Stop
	SSGFlash:
		TNT1 A 0 A_PlaySound("weapons/sawidle",5)
		SAWS RQPON 1
		SAWS D 20
		SAWS NOPQR 1
		Stop
	SlideKickingStart:
		TNT1 A 0 A_PlaySound("weapons/sawidle",5)
		SAWS RQPON 1
		SAWS D 18
	SlideKickingEnd:
		SAWS D 6
		SAWS NOPQR 1
		Stop
	Spawn:
		CSAW A -1
		Stop
	Deselect:
		SAWS RQPONDCBA 1
	DeselectEnd:
		TNT1 A 0 A_SetCrosshair(0)
		TNT1 A 0 A_StopSound(5)
		TNT1 A 0 A_StopsoundEx("Weapon")
		TNT1 AAAAAAAAAAAAAAAAAAAAAAAA 0 A_Lower
		Wait
	Select:
		TNT1 A 0 A_JumpIfInventory("SawGas",1,"Select2")
		TNT1 AAAAAAAAAA 0 A_Raise
		Goto SelectAnimation
	Select2:
		TNT1 AAAAAAAAAA 0 A_Raise
		Goto SelectAnimation2
	SelectAnimation2:
		SAWS ABCDDDDD 1
		TNT1 A 0 A_Jump(125,2)
		TNT1 A 0 A_JumpIfInventory("ChainsawJamming",1,"JammedAnimation")
		TNT1 A 0 A_TakeInventory("ChainsawJamming",1)
		SAWG A 0 A_PlaySound("DSSAWZIP",1)
		SAWG A 0 A_GunFlash
		SAWS EFGHIJKLM 1
		TNT1 A 0 A_playsound("weapons/chainsaw/draw",3)
		SAWT AB 2
		SAWS NOPQR 1
		Goto Ready
	SelectAnimation:
		SAWS ABCDNOPQR 1
		Goto Ready
	v1Ready:
	Ready:
		TNT1 A 0 A_JumpIfInventory("Kicking",1,"DoKick")
		TNT1 A 0 A_JumpIfInventory("SawGas",1,1)
		Goto NotReady
		SAWG AB 3 A_WeaponReady(WRF_ALLOWRELOAD)
		TNT1 A 0 A_PlaySound("weapons/sawidle",5)
		SAWG CD 3 A_WeaponReady(WRF_ALLOWRELOAD)
		TNT1 A 0 A_JumpIfInventory("istacticalclass",1,"Ready")		//R28.2 added
		TNT1 A 0 A_TakeInventory("Stamina",0)						//R28.2 added
		Loop
	NotReady:
		SAWG A 5 A_WeaponReady
		TNT1 A 0 A_JumpIfInventory("Kicking",1,"DoKick")
		TNT1 A 0 A_JumpIfInventory("SawGas",1,"Select")
		Loop
	Fire:
//		TNT1 A 0 A_PlaySound("weapons/chainsaw/start",3)		//R11.3 removed
		TNT1 A 0 A_JumpIfInventory("SawGas",1,1)
		Goto FinishFire
		SAWF AB 1
	Hold:
		TNT1 A 0 A_PlaySound("weapons/chainsaw/loop",1)
		TNT1 A 0 A_AlertMonsters
		TNT1 A 0 A_TakeInventory("SawGas",1)
		SAWF C 1 Offset(0,36) A_FireCustomMissile("SawDamage",10,0,0,0)
		SAWF C 1 Offset(-1,34) A_FireCustomMissile("SawDamage",-10,0,0,0)
		SAWF C 1 Offset(0,35) A_FireCustomMissile("SawDamage",10,0,0,0)
		SAWF D 1 Offset(1,33) A_FireCustomMissile("SawDamage",-10,0,0,0)
		SAWF C 1 Offset(0,35) A_FireCustomMissile("SawDamage",10,0,0,0)
		SAWF C 1 Offset(-1,34) A_FireCustomMissile("SawDamage",-10,0,0,0)
		SAWF D 1 Offset(0,37) A_FireCustomMissile("SawDamage",10,0,0,0)
		SAWF D 1 Offset(1,34) A_FireCustomMissile("SawDamage",-10,0,0,0)
		TNT1 A 0 A_Jump(5,"Jammed")
		TNT1 A 0 A_ReFire
	FinishFire:
		SAWF BA 1
		Goto Ready
	AltFire:
		TNT1 A 0 A_Jumpifinventory("64SuperShotgunTactical",1,2)
		TNT1 A 0 A_Print("You Need a Super Shotgun",1)
		Goto Ready
		TNT1 A 0 A_Jumpifinventory("SSGAmmo",1,3)
		TNT1 A 0 A_PlaySound("DRYFIRE")
		TNT1 A 0 A_Print("Super Shotgun Needs Reloading",1)
		Goto Ready
		TNT1 A 0 A_Jumpifinventory("SSGAmmo",2,1)
		Goto AltFire2
		TNT1 A 0 A_Jumpifinventory("Shell",2,1)
		TNT1 A 10 A_GunFlash("SSGFlash")
		OSSG ABCD 1 Offset(0,50)
		TNT1 A 0 A_SetAngle(angle + random(-10,10))
		TNT1 AAAA 0 A_FireCustomMissile("GunFireSmoke",0,0,0,0,0,0)
		TNT1 AA 0 A_SpawnItemEx("PlayerMuzzle1",30,5,40)
		TNT1 AAAAAAAAAA 0 A_FireCustomMissile("Tracer",random(-6,6),0,-1,-12,0,random(-5,5))
		TNT1 A 0 A_FireBullets(8.5,6.5,20,6,"shotpuff")
		TNT1 A 0 A_TakeInventory("SSGAmmo",2)
		TNT1 A 0 A_PlaySound("SSGFIRE",1)
		TNT1 A 0 A_AlertMonsters
		TNT1 A 0 A_ZoomFactor(0.75)
		OSSF AB 1 Bright
		TNT1 A 0 A_ZoomFactor(1.0)
		SSGF C 3 A_SetPitch(pitch-4)
		SSGF DE 1 A_SetPitch(pitch+2.6)
		SSGF F 1 A_SetPitch(pitch+1.2)
		SSGG A 1 A_SetPitch(pitch+0.8)
		SSGG A 1 A_SetPitch(pitch+0.4)
		OSSG DCBA 1 Offset(0,50)
		Goto Ready
	AltFire2:
		TNT1 A 10 A_GunFlash("SSGFlash")
		OSSG ABCD 1 Offset(0,50)
		TNT1 A 0 A_SetAngle(angle + random(-6,6))
		TNT1 AA 0 A_FireCustomMissile("GunFireSmoke",0,0,0,0,0,0)
		TNT1 A 0 A_SpawnItemEx("PlayerMuzzle1",30,5,40)
		TNT1 AAAA 0 A_FireCustomMissile("Tracer",random(-6,6),0,-1,-12,0,random(-5,5))
		TNT1 A 0 A_FireBullets(7.5,5.5,10,6,"ShotgunPuff")
		TNT1 A 0 A_TakeInventory("SSGAmmo",1)
		TNT1 A 0 A_PlaySound("SGFIRE",1)
		TNT1 A 0 A_AlertMonsters
		TNT1 A 0 A_ZoomFactor(0.95)
		OSSU AB 1 Bright
		TNT1 A 0 A_ZoomFActor(1.0)
		SSGF E 3 A_SetPitch(pitch-2)
		TNT1 A 0 Offset(-4,32)
		SSGF E 1 A_SetPitch(pitch+1.3)	//r31 was frame E
		TNT1 A 0 Offset(-3,32)
		SSGF F 1 A_SetPitch(pitch+0.6)
		TNT1 A 0 Offset(-2,32)
		SSGG A 1 A_SetPitch(pitch+0.4)
		TNT1 A 0 Offset(-1,32)
		SSGG A 1 A_SetPitch(pitch+0.2)
		TNT1 A 0 Offset(0,32)
		OSSG DCBA 1 Offset(0,32)
		Goto Ready
	Reload:
		TNT1 A 0 A_JumpIfInventory("ChainsawJamming",1,"Unjammed")
		SAWS RQPON 1
		SAWS D 1
		SAWG A 0 A_PlaySound("DSSAWZIP",1)
		SAWS EFGHHGFE 1
		TNT1 A 0 A_playsound("weapons/chainsaw/draw",3)
		SAWT AB 2 
		SAWS NOPQR 1
		Goto Ready
	Jammed:
		TNT1 A 0 A_GiveInventory("ChainsawJamming",1)
		TNT1 A 0 A_Playsound("weapons/chainsaw/stop",3)
		SAWS RQPON 1
	JammedAnimation:
		SAWS D 1 A_WeaponReady(WRF_ALLOWRELOAD|WRF_NOFIRE)
		SAWG A 0 A_PlaySound("DSSAWZIP",1)
		SAWS EFGHHGFE 1 A_WeaponReady(WRF_ALLOWRELOAD|WRF_NOFIRE)
		SAWG A 0 A_PlaySound("DSSAWZIP",1)
		SAWS EFGHHGFE 2	A_WeaponReady(WRF_ALLOWRELOAD|WRF_NOFIRE)
		SAWG A 0 A_PlaySound("DSSAWZIP",1)
		SAWS EFGHHGFE 1 A_WeaponReady(WRF_ALLOWRELOAD|WRF_NOFIRE)
		SAWG A 0 A_PlaySound("DSSAWZIP",1)
		SAWS EFGHHGFE 2 A_WeaponReady(WRF_ALLOWRELOAD|WRF_NOFIRE)
		SAWS D 5 A_playsound("DSSAWDIE",3)
		Loop
	Unjammed:
		TNT1 A 0 A_TakeInventory("ChainsawJamming",1)
		SAWG A 0 A_PlaySound("DSSAWZIP",1)
		SAWS EFGHIJKLM 1
		TNT1 A 0 A_playsound("weapons/chainsaw/draw",3)
		SAWT AB 2 
		SAWS NOPQR 1
		Goto Ready
	}
}
