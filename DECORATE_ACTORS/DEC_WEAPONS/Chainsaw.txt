Actor 64Chainsaw : Weapon	//RETAIN CLASSIC LOOK
{
Scale 0.82
Weapon.SelectionOrder 2200
Inventory.PickupMessage "$GOTCHAINSAW"
Weapon.UpSound "null"
+WEAPON.AXEBLOOD
Tag "$TAG_CHAINSAW"
VisibleToPlayerClass "64DoomPlayer"
Inventory.ForbiddenTo "Tactical64DoomPlayer","Tactical64DoomPlayer_p","Modern"
+INVENTORY.RESTRICTABSOLUTELY
	States
	{
	Spawn:
		CWAS A -1
		Stop
	Select:
		TNT1 A 0 A_PlaySoundEx("weapons/sawup",SOUNDSLOT7,0)
		WASG C 1 A_Raise
		Wait
	Ready:
		WASG CD 5 A_WeaponReady
		TNT1 A 0 A_PlaySound("weapons/sawidle",5)
		Loop
	Fire:
		WASG HI 1
	Hold:
		TNT1 A 0 A_PlaySound("weapons/chainsaw/loop",1)
		WASG ABEF 2 A_FireCustomMissile("SawDamage",0,0,0,0)
		TNT1 A 0 A_ReFire
		TNT1 A 0 A_PlaySound("weapons/chainsaw/Stop",3)
		TNT1 A 0 A_PlaySound("weapons/chainsaw/Stop",1)
		WASG IH 1
		Goto Ready
	Deselect:
		WASG C 1 A_Lower
		Wait
	}
}
