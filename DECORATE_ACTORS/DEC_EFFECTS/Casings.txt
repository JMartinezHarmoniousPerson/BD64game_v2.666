ACTOR EmptyBrass
{
Scale 0.15
Height 2
Radius 2
Speed 6
Mass 4
SeeSound "weapons/casing"
-NOGRAVITY
-DROPOFF
+WINDTHRUST
+CLIENTSIDEONLY
+MOVEWITHSECTOR
+MISSILE
+NOBLOCKMAP
+NOTELEPORT
+FORCEXYBILLBOARD
+NOTDMATCH
+GHOST
+FLOORCLIP
+THRUACTORS
+ISMONSTER		//INFRARED
+DOOMBOUNCE
BounceFactor 0.3
	States
	{
	Spawn:
		CAS3 A 0 NoDelay A_Jump(128,2)
		TNT1 A 0 A_SetScale(0-ScaleX,ScaleY)		//FLIP IMAGE HORZ
		TNT1 A 0 A_JumpIf(waterlevel > 1,"Splash")
		CAS3 FGHABCDE 3 A_CheckFloor("Death")
		WAIT
	Death:
		TNT1 A 0 ACS_NamedExecuteAlways("LowBloodCheckerACS")//Check Effects
		LCPJ A 0 A_Jump(256,"Rest1","Rest2","Rest3")
		Goto Rest1
	Rest1:
		CAS3 I 400
		TNT1 A 0 A_JumpIfInventory("LowGraphicsMode",1,"Stoping")
		Loop
	Rest2:
		CAS3 J 400
		TNT1 A 0 A_JumpIfInventory("LowGraphicsMode",1,"Stoping")
		Loop
	Rest3:
		CAS3 K 400
		TNT1 A 0 A_JumpIfInventory("LowGraphicsMode",1,"Stoping")
		Loop
	Splash:
		TNT1 A 1 A_SpawnItemEx("UnderwaterEmptyBrass")
		Stop
	Stoping:
		TNT1 A 1
		Stop
	}
}
