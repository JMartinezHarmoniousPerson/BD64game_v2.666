ACTOR DetectFloorBullet		//r31 checked and working
{
Mass 1	//R31 added
scale 0.4
radius 1
height 1
Alpha 0.9	//R31 WAS 0.7
Renderstyle Translucent
-DONTSPLASH
+MISSILE
+CLIENTSIDEONLY
+NOTELEPORT
+NOBLOCKMAP
+FORCEXYBILLBOARD
+NODAMAGETHRUST
+MOVEWITHSECTOR
+DOOMBOUNCE
BounceFactor 0.01
	States
	{
	Spawn:
		TNT1 A 0 NODELAY A_JumpIf(CallACS("Flat_Decal_Remover")==1,"Off")			//r31
//		TNT1 A 1 ACS_ExecuteAlways(17001,0)		//R31 REMOVED
//	On:																		//R31 REMOVED
		TNT1 A 1 ThrustThingZ(0,-10,0,1)	//R31 28.7.22	WAS 0 TIC
		TNT1 AA 1 A_CHECKFLOOR("death")											//R31 added floor check
		STOP
	Off:
		TNT1 A 0 A_FadeOut(1)
		Stop
	Death:
		XXX1 A 2000
		Stop
	}
}
