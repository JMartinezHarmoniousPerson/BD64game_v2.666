ACTOR DetectFloorBullet		//r31 checked and working
{
Mass 1	//R31 added
scale 0.4
radius 12
height 1
Alpha 0.9	//R31 WAS 0.7
Renderstyle Translucent
+FLOORCLIP
-DONTSPLASH
+MISSILE
+CLIENTSIDEONLY
+NOTELEPORT
+NOBLOCKMAP
+FORCEXYBILLBOARD
+NODAMAGETHRUST
+MOVEWITHSECTOR
+DOOMBOUNCE
BounceFactor 0.01
	States
	{
	Spawn:
		TNT1 A 0 NoDelay A_JumpIf(CallACS("Flat_Decal_Remover")==1,"Off")	//r31 revised
		TNT1 A 0 A_JumpIf(CallACS("FloorCheck")==FALSE,"Off")	//replaces A_CheckFloor
		TNT1 A 0 ThrustThingZ(0,10,1,0)
	Death:
		XXX1 A 2000
		XXX1 A 1 A_FadeOut(0.01)
		Wait
	Off:
		TNT1 A 0
		Stop
	}
}
