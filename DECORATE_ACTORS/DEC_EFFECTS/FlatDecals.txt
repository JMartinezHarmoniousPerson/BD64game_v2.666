ACTOR DetectFloorBullet
{
Mass 1
Gravity 1
scale 0.4
radius 12
height 1
Alpha 0.9
Renderstyle Translucent
+FLOORCLIP
+DONTSPLASH
+CLIENTSIDEONLY
+NOTELEPORT
+NOBLOCKMAP
+FORCEXYBILLBOARD
+NODAMAGETHRUST
+MOVEWITHSECTOR
	States
	{
	Spawn:
		TNT1 A 0 NoDelay A_JumpIf(CallACS("Flat_Decal_Remover")==1,"Off")
		TNT1 AA 1 A_CheckFloor("Death")
		Goto Off
	Death:
		XXX1 A 2000
		XXX1 A 1 A_FadeOut(0.01)
		Wait
	Off:
		TNT1 A 1
		Stop
	}
}
